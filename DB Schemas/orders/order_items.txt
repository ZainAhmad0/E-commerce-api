CREATE TABLE ORDER_ITEMS (
ORDER_ID VARCHAR(100) NOT NULL,
PRODUCT_ID VARCHAR(100) NOT NULL,
SELLER_ID VARCHAR(100) NOT NULL,
CREATED_AT TIMESTAMP,
PRIMARY KEY (ORDER_ID,PRODUCT_ID),
CONSTRAINT FK_ORDER_ID FOREIGN KEY (ORDER_ID) REFERENCES ORDER_DETAILS (ID) ON
DELETE
	NO ACTION ON
	UPDATE
		NO ACTION,
	CONSTRAINT FK_PRODUCT_ORDER FOREIGN KEY (PRODUCT_ID) REFERENCES PRODUCT (ID) ON
DELETE
	NO ACTION ON
	UPDATE
		NO ACTION,
	CONSTRAINT FK_ORDER_SELLER_ID FOREIGN KEY (SELLER_ID) REFERENCES INVENTORY.PUBLIC.USER_INFO (ID) ON
				DELETE
					NO ACTION ON
					UPDATE
						NO ACTION);
